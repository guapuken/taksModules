# Ejemplo de uso

    	const [activeIndex, setActiveIndex] = useState( 1 );
    	function getInfoToShow( ) {
    		switch (activeIndex) {
    			case 1:return <div>Soy detalles</div>;
    			case 2:return <div>Soy comentarios y precios</div>;
    			case 3:return <div>Soy Data</div>;
    			case 4:return <div>Soy Opciones extras</div>;
    			case 5:return <div>Soy Dudas del sitio</div>;
    			default:break;
    		}
    	}
        function getMenus( ) {
    		return [
    			{ title: 'detalles', index: 1, },
    			{ title: 'Comentarios y Precios', index: 2, },
    			{ title: 'Data', index: 3, },
    		];
    	}
    	return (
            <CardsGenericas
                activeIndex={activeIndex}
                setActiveIndex={setActiveIndex}
                modo="Light"
                menus={getMenus( )}
                initialView={ <div className='InitialView'> Hola </div> }
            >
                {getInfoToShow( )}
            </CardsGenericas>
    	);
